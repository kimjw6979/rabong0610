import pandas as pd
import os
import random
import sys # 상단에 import sys 추가

def balance_teams():
    # 1. 파일 경로 설정 (수정됨: 실행 파일 위치를 정확히 찾도록 변경)
    if getattr(sys, 'frozen', False):
        # 실행 파일(.exe)로 실행될 때
        current_dir = os.path.dirname(sys.executable)
    else:
        # 파이썬 코드로 실행될 때
        current_dir = os.path.dirname(os.path.abspath(__file__))
    score_path = os.path.join(current_dir, '분기에버.xlsx')
    attend_path = os.path.join(current_dir, '참석자명단.xlsx')
    
    try:
        # 1. 파일 로드
        scores_df = pd.read_excel(score_path, header=6)
        attend_df = pd.read_excel(attend_path)
        
        # 2. 이름 정제 및 여성(W) 인식 로직
        def parse_name(name):
            name_str = str(name).strip().replace(' ', '')
            is_female = name_str.endswith('W')
            clean_name = name_str[:-1] if is_female else name_str
            return clean_name, is_female

        parsed_data = scores_df['이 름'].apply(parse_name).tolist()
        scores_df['이름_cleaned'] = [x[0] for x in parsed_data]
        scores_df['is_female'] = [x[1] for x in parsed_data]
        
        attend_df['이름_cleaned'] = attend_df['이름'].astype(str).str.replace(' ', '')
        
        # 3. 데이터 병합
        df = pd.merge(scores_df, attend_df, on='이름_cleaned')
        
        # 4. 참석자(O) 필터링
        df['참석'] = df['참석'].astype(str).str.strip().str.upper()
        attendees = df[df['참석'] == 'O'].copy()
        
        if attendees.empty:
            print("참석자(O)가 없습니다. 참석자명단.xlsx를 확인해주세요.")
            return

        print(f"총 참석자: {len(attendees)}명")
        
        females_list = attendees[attendees['is_female']]['이 름'].tolist()
        if females_list:
            print(f"인식된 여성 회원({len(females_list)}명): {', '.join(females_list)}")

        # 5. 팀 개수 입력
        num_teams = int(input("\n나눌 팀 개수(2~3)를 입력하세요: "))
        
        # 6. 인원수 맞추기 (나머지 인원 제외)
        remainder = len(attendees) % num_teams
        target_size = len(attendees) // num_teams 
        
        if remainder != 0:
            attendees = attendees.sort_values(by='적용에버', ascending=True)
            excluded_people = attendees.iloc[:remainder]
            
            print(f"\n[알림] 각 팀 인원을 {target_size}명으로 맞추기 위해 가장 점수가 낮은 {remainder}명을 제외합니다.")
            for _, person in excluded_people.iterrows():
                print(f" -> 제외(개별 관리): {person['이 름']} (적용에버: {person['적용에버']})")
            
            df_for_teams = attendees.iloc[remainder:].copy()
        else:
            df_for_teams = attendees.copy()
            
        # 7. 여성/남성 분리 후 정렬
        females = df_for_teams[df_for_teams['is_female'] == True].sort_values(by='적용에버', ascending=False)
        males = df_for_teams[df_for_teams['is_female'] == False].sort_values(by='적용에버', ascending=False)

        # 8. 팀 나누기 로직
        teams = [{'members': [], 'sum': 0, 'f_count': 0} for _ in range(num_teams)]
        
        # [1단계] 여성 분배
        for _, row in females.iterrows():
            min_f = min(t['f_count'] for t in teams)
            candidate_teams = [t for t in teams if t['f_count'] == min_f]
            target_team = min(candidate_teams, key=lambda x: x['sum'])
            
            target_team['members'].append(row)
            target_team['sum'] += row['적용에버']
            target_team['f_count'] += 1
            
        # [2단계] 남성 분배
        for _, row in males.iterrows():
            candidate_teams = [t for t in teams if len(t['members']) < target_size]
            target_team = min(candidate_teams, key=lambda x: x['sum'])
            
            target_team['members'].append(row)
            target_team['sum'] += row['적용에버']
            
        # 9. 팀 구성 결과 출력
        print(f"\n=== {num_teams}개 팀 구성 결과 ===")
        for i in range(num_teams):
            names = [member['이 름'] for member in teams[i]['members']]
            avg_score = teams[i]['sum'] / len(teams[i]['members']) if len(teams[i]['members']) > 0 else 0
            
            print(f"\n[팀 {i+1}] (인원: {len(names)}명, 여성: {teams[i]['f_count']}명)")
            print(f"구성원: {', '.join(names)}")
            print(f"팀 평균 에버: {avg_score:.2f}")

        # 10. 레인 배정 로직
        active_count = sum(len(t['members']) for t in teams)
        
        # 18명 이하: 6레인 / 19명 이상: 8레인
        if active_count <= 18:
            num_lanes = 6
        else:
            num_lanes = 8
            
        lanes = [[] for _ in range(num_lanes)]
        
        # [case 1] 2팀으로 나눌 때
        if num_teams == 2:
            print(f"\n=== 🎲 랜덤 레인 배정 결과 (총 {num_lanes}개 레인 | [팀1]: 홀수, [팀2]: 짝수) ===")
            odd_lane_indices = [idx for idx in range(num_lanes) if idx % 2 == 0]  # 0, 2, 4, 6 (1, 3, 5, 7번 레인)
            even_lane_indices = [idx for idx in range(num_lanes) if idx % 2 != 0] # 1, 3, 5, 7 (2, 4, 6, 8번 레인)
            
            # 🔥 핵심 변경 부분: 한 테이블(예:1번,2번)의 인원 편차를 막기 위해 짝수 레인은 배정 순서를 뒤집음
            # 팀1이 1번,3번에 3명씩 들어갈 때, 팀2는 8번,6번에 3명씩 들어가서 테이블당 인원을 똑같이 맞춥니다.
            even_lane_indices.reverse()
            
            # 팀 1 회원 분배
            t1_players = [f"[팀1] {m['이 름']}" for m in teams[0]['members']]
            random.shuffle(t1_players)
            for idx, player in enumerate(t1_players):
                target_lane = odd_lane_indices[idx % len(odd_lane_indices)]
                lanes[target_lane].append(player)
                
            # 팀 2 회원 분배 (뒤집힌 순서로 들어감)
            t2_players = [f"[팀2] {m['이 름']}" for m in teams[1]['members']]
            random.shuffle(t2_players)
            for idx, player in enumerate(t2_players):
                target_lane = even_lane_indices[idx % len(even_lane_indices)]
                lanes[target_lane].append(player)
                
        # [case 2] 3팀으로 나눌 때
        elif num_teams == 3:
            print(f"\n=== 🎲 랜덤 레인 배정 결과 (총 {num_lanes}개 레인 | 팀별 연속 레인 배정) ===")
            team_lane_map = {0: [], 1: [], 2: []}
            
            for i in range(num_lanes):
                if i < 6:
                    team_lane_map[i // 2].append(i)
                else:
                    team_lane_map[i - 6].append(i)
                    
            for t_idx in range(3):
                t_players = [f"[팀{t_idx+1}] {m['이 름']}" for m in teams[t_idx]['members']]
                random.shuffle(t_players)
                assigned_lanes = team_lane_map[t_idx]
                
                for idx, player in enumerate(t_players):
                    target_lane = assigned_lanes[idx % len(assigned_lanes)]
                    lanes[target_lane].append(player)
            
        # 결과 최종 출력
        for i in range(num_lanes):
            print(f"레인 {i+1} ({len(lanes[i])}명): {', '.join(lanes[i])}")

    except Exception as e:
        print(f"오류 발생: {e}")

if __name__ == "__main__":
    balance_teams()
    # 결과 확인 후 엔터를 누를 때까지 창을 유지합니다.
    input("\n모든 결과 확인이 끝나셨나요? 엔터 키를 누르면 프로그램이 종료됩니다.")
    balance_teams()
